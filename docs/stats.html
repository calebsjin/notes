Let a \((k+1)\times (k+1)\) matrix \({\bf M}\) be partitioned in to a block form: \[ {\bf M}= \left[\begin{array}{cc}{\bf A}&{\bf b}\\{\bf b}^{{\top}}&c\end{array}\right], \] where \({\bf A}\) is \(k\times k\) matrix, \({\bf b}\) is \(k\times 1\) vector, and \(c\) is a scalar. Then \[ {\bf M}^{-1} = \left[\begin{array}{cc}{\bf A}^{-1}+m^{-1}{\bf A}^{-1}{\bf b}{\bf b}^{{\top}}{\bf A}^{-1}&-m^{-1}{\bf A}^{-1}{\bf b}\\-m^{-1}{\bf b}^{{\top}}{\bf A}^{-1}&m^{-1}\end{array}\right], \] where \(m=c - {\bf b}^{{\top}}{\bf A}^{-1}{\bf b}\).

Let \({\bf A}= {\bf X}_{\boldsymbol \gamma}^{{\top}}{\bf X}_{\boldsymbol \gamma}+ \lambda^{-1}{\bf I}_{|{\boldsymbol \gamma}|}\), \({\bf b}= {\bf X}_{\boldsymbol \gamma}^{{\top}}{\bf x}_j\), and \(c = {\bf x}_j^{{\top}}{\bf x}_j + \lambda^{-1}\). \[\begin{eqnarray*} \hat{\boldsymbol \beta}_{{\boldsymbol \gamma}^{+j}} &=& ({\bf X}_{{\boldsymbol \gamma}^{+j}}^{{\top}}{\bf X}_{{\boldsymbol \gamma}^{+j}}+\lambda^{-1}{\bf I}_{|{{\boldsymbol \gamma}^{+j}}|})^{-1}{\bf X}_{{\boldsymbol \gamma}^{+j}}^{{\top}}{\bf y}\\ &=&\left[\begin{array}{cc} {\bf X}_{\boldsymbol \gamma}^{{\top}}{\bf X}_{\boldsymbol \gamma}+ \lambda^{-1}{\bf I}_{|{\boldsymbol \gamma}|} & {\bf X}_{\boldsymbol \gamma}^{{\top}}{\bf x}_j \\ {\bf x}_j^{{\top}}{\bf X}_{\boldsymbol \gamma}& {\bf x}_j^{{\top}}{\bf x}_j + \lambda^{-1} \end{array}\right]^{-1}\left[\begin{array}{c}{\bf X}_{\boldsymbol \gamma}^{{\top}}\\{\bf x}_j^{{\top}}\end{array}\right]{\bf y}\\ &=&\left[\begin{array}{cc}{\bf A}&{\bf b}\\{\bf b}^{{\top}}&c\end{array}\right]^{-1}\left[\begin{array}{c}{\bf X}_{\boldsymbol \gamma}^{{\top}}\\{\bf x}_j^{{\top}}\end{array}\right]{\bf y}\\ &=&\left[\begin{array}{cc}{\bf A}^{-1}+m^{-1}{\bf A}^{-1}{\bf b}{\bf b}^{{\top}}{\bf A}^{-1}&-m^{-1}{\bf A}^{-1}{\bf b}\\-m^{-1}{\bf b}^{{\top}}{\bf A}^{-1}&m^{-1}\end{array}\right]^{-1}\left[\begin{array}{c}{\bf X}_{\boldsymbol \gamma}^{{\top}}\\{\bf x}_j^{{\top}}\end{array}\right]{\bf y} \end{eqnarray*}\] Therefore, \[\begin{eqnarray*} m &=& {\bf x}_j{\bf x}_j^{{\top}}+\lambda^{-1} - {\bf x}_j^{{\top}}{\bf X}_{\boldsymbol \gamma}({\bf X}_{\boldsymbol \gamma}^{{\top}}{\bf X}_{\boldsymbol \gamma}+ \lambda^{-1}{\bf I}_{|{\boldsymbol \gamma}|})^{-1}{\bf X}_{\boldsymbol \gamma}{\bf x}_j\\ &=& \lambda^{-1} + {\bf x}_j({\bf I}_n - {\bf X}_{\boldsymbol \gamma}({\bf X}_{\boldsymbol \gamma}^{{\top}}{\bf X}_{\boldsymbol \gamma}+ \lambda^{-1}{\bf I}_{|{\boldsymbol \gamma}|})^{-1}{\bf X}_{\boldsymbol \gamma}^{{\top}}){\bf x}_j^{{\top}}\\ &=& \lambda^{-1} + {\bf x}_j({\bf I}_n - {\bf P}_{\boldsymbol \gamma}){\bf x}_j^{{\top}}\\ \hat\beta_j &=& (-m^{-1}{\bf b}^{{\top}}{\bf A}^{-1}{\bf X}_{\boldsymbol \gamma}^{{\top}} + m^{-1}{\bf x}_j^{{\top}}){\bf y}\\ &=&m^{-1}({\bf x}_j^{{\top}} - {\bf b}^{{\top}}{\bf A}^{-1}{\bf X}_{\boldsymbol \gamma}^{{\top}}){\bf y}\\ &=& m^{-1}({\bf x}_j^{{\top}} - {\bf x}_j^{{\top}}{\bf X}_{\boldsymbol \gamma}({\bf X}_{\boldsymbol \gamma}^{{\top}}{\bf X}_{\boldsymbol \gamma}+ \lambda^{-1}{\bf I}_{|{\boldsymbol \gamma}|})^{-1}{\bf X}_{\boldsymbol \gamma}^{{\top}}){\bf y}\\ &=& m^{-1}{\bf x}_j^{{\top}}({\bf I}_n - {\bf X}_{\boldsymbol \gamma}({\bf X}_{\boldsymbol \gamma}^{{\top}}{\bf X}_{\boldsymbol \gamma}+ \lambda^{-1}{\bf I}_{|{\boldsymbol \gamma}|})^{-1}{\bf X}_{\boldsymbol \gamma}^{{\top}}){\bf y}\\ &=& m^{-1}{\bf x}_j^{{\top}}({\bf y}- {\bf X}_{\boldsymbol \gamma}({\bf X}_{\boldsymbol \gamma}^{{\top}}{\bf X}_{\boldsymbol \gamma}+ \lambda^{-1}{\bf I}_{|{\boldsymbol \gamma}|})^{-1}{\bf X}_{\boldsymbol \gamma}^{{\top}}{\bf y})\\ &=& m^{-1}{\bf x}_j^{{\top}}({\bf y}- {\bf X}_{\boldsymbol \gamma}\hat{\boldsymbol \beta}_{\boldsymbol \gamma})\\ &=& \frac{{\bf x}_j^{{\top}}\hat{\boldsymbol \epsilon}_{\boldsymbol \gamma}}{\lambda^{-1} + {\bf x}_j({\bf I}_n - {\bf P}_{\boldsymbol \gamma}){\bf x}_j^{{\top}}}, \end{eqnarray*}\] where \(\hat{\boldsymbol \beta}_{\boldsymbol \gamma}= ({\bf X}_{\boldsymbol \gamma}^{{\top}}{\bf X}_{\boldsymbol \gamma}+\lambda^{-1}{\bf I}_{|{\boldsymbol \gamma}|})^{-1}{\bf X}_{\boldsymbol \gamma}^{{\top}}{\bf y}\), \(\hat{\boldsymbol \epsilon}_{\boldsymbol \gamma}= {\bf y}-{\bf X}_{\boldsymbol \gamma}\hat{\boldsymbol \beta}_{\boldsymbol \gamma}\), and \({\bf P}_{\boldsymbol \gamma}= {\bf X}_{{\boldsymbol \gamma}}({\bf X}_{{\boldsymbol \gamma}}^{{\top}}{\bf X}_{{\boldsymbol \gamma}}+\lambda^{-1}{\bf I}_{|{{\boldsymbol \gamma}}|})^{-1}{\bf X}_{{\boldsymbol \gamma}}^{{\top}}\).
